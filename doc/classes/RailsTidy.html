129:   def format_content(content)
130:     # add doctype if not found
131:     validable = ""
132:     unless content.match(/<!DOCTYPE/)
133:       validable << "<!DOCTYPE html PUBLIC \"-//W3C//DTD XHTML 1.0 Transitional//EN\" " +
134:         " \"http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd\">"
135:     end
136: 
137:     # add html, head, title and body

142:     end
143: 
144:     # replace start_form_tag and end_form_tag with <form> and </form>

106:     # remove errors that are to be ignored
107:     @tidy.errors.delete_if do |error|
108:       IGNORED.inject(false) { |b,r| b ||= error.message.match(r) }
